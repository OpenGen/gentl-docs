35 double mh_accept_prob(double model_log_weight, double proposal_forward_score, double proposal_backward_score) {
36  return std::min(1.0, std::exp(model_log_weight + proposal_backward_score - proposal_forward_score));
37 }

46 template<typename ModelTrace, typename ProposalTrace, typename Proposal,
47  typename ProposalParams, typename RNGType>
48 #ifdef __cpp_concepts
49 requires
50  SimulatableGenerativeFunction<Proposal,ModelTrace,ProposalParams,ProposalTrace> &&

54 bool mh(RNGType &rng, ModelTrace &model_trace,
55  const Proposal &proposal, ProposalParams &proposal_params, ProposalTrace &proposal_trace,
56  bool precompute_gradient = false) {
57  proposal(model_trace).simulate(rng, proposal_params, SimulateOptions(), proposal_trace);
58  double proposal_forward_score = proposal_trace.score();
59  const auto &forward_constraints = proposal_trace.choices(gentl::selection::all);
60  double model_log_weight = model_trace.update(
61  rng, gentl::change::no_change,
62  forward_constraints,
63  UpdateOptions().save(true).precompute_gradient(precompute_gradient));
64  const auto& backward_constraints = model_trace.backward_constraints();
65  auto proposal_backward_score = proposal(model_trace).assess(rng, proposal_params, backward_constraints);
66  double prob_accept = mh_accept_prob(model_log_weight, proposal_forward_score, proposal_backward_score);
67  bool accept = std::bernoulli_distribution{prob_accept}(rng);
68  if (!accept)
69  model_trace.revert();
70  return accept;
71 }

77 template <typename ChoiceBuffer, typename RNGType>
78 double mala_propose_values(ChoiceBuffer& proposal, const ChoiceBuffer& current,
79  const ChoiceBuffer& gradient, double tau, RNGType& rng) {
80  double stdev = std::sqrt(2 * tau);
81  static std::normal_distribution<double> standard_normal {0.0, 1.0};
82  static double logSqrt2Pi = 0.5*std::log(2*M_PI);
83  double log_density = (current.cend() - current.cbegin()) * (-std::log(stdev) - logSqrt2Pi);
84 
85  // first compute the mean in-place
86  proposal = current + (tau * gradient);
87 
88  // then sample new values (over-writing the mean), and finish computing the log density
89  for (auto proposal_it = proposal.begin(); proposal_it != proposal.end(); proposal_it++) {
90  double standard_normal_increment = standard_normal(rng);
91  *proposal_it += (standard_normal_increment * stdev);
92  log_density += -0.5 * (standard_normal_increment * standard_normal_increment);
93  }
94  return log_density;
95 }

97 template <typename ChoiceBuffer>
98 double mala_assess(const ChoiceBuffer& proposed, const ChoiceBuffer& current,
99  const ChoiceBuffer& gradient, double tau, ChoiceBuffer& storage) {
100  double stdev = std::sqrt(2 * tau);
101  static double logSqrt2Pi = 0.5*std::log(2*M_PI);
102  double log_density = (current.cend() - current.cbegin()) * (-std::log(stdev) - logSqrt2Pi);
103 
104  ChoiceBuffer& proposal_mean = storage; // rename it
105  proposal_mean = current + (tau * gradient);
106 
107  auto proposal_mean_it = proposal_mean.cbegin();
108  for (auto proposed_it = proposed.cbegin(); proposed_it != proposed.cend(); proposed_it++) {
109  double standard_normal_increment = (*proposed_it - *(proposal_mean_it++)) / stdev;
110  log_density += -0.5 * (standard_normal_increment * standard_normal_increment);
111  }
112  return log_density;
113 }

117 template<typename Trace, typename Selection, typename RNG, typename ChoiceBuffer>
118 #ifdef __cpp_concepts
119 requires TopLevelGradientBasedChoiceUpdates<Trace,Selection,ChoiceBuffer> && Revertible<Trace>
120 #endif
121 bool mala(Trace &trace, const Selection &selection, double tau,
122  ChoiceBuffer &storage1, ChoiceBuffer& storage2, RNG &rng) {
123 
124  // NOTE: these buffers are only valid up until the next call to update.
125  ChoiceBuffer& proposed_values = storage1;
126  double forward_log_density = mala_propose_values(proposed_values, trace.choices(selection),
127  trace.choice_gradient(selection), tau, rng);
128  double log_weight = trace.update(
129  rng, gentl::change::no_change, proposed_values,
130  UpdateOptions().save(true).precompute_gradient(true));
131 
132  // compute backward log density
133  const auto& previous_values = trace.backward_constraints();
134  double backward_log_density = mala_assess(previous_values, proposed_values,
135  trace.choice_gradient(selection), tau,
136  storage2);
137 
138  double prob_accept = std::min(1.0, std::exp(log_weight + backward_log_density - forward_log_density));
139  bool accept = std::bernoulli_distribution{prob_accept}(rng);
140  if (!accept)
141  trace.revert();
142  return accept;
143 }

150 template <typename ChoiceBufferType, typename RNGType>
151 void sample_momenta(ChoiceBufferType& momenta, RNGType& rng) {
152  static std::normal_distribution<double> standard_normal{0.0, 1.0};
153  for (auto& momentum : momenta)
154  momentum = standard_normal(rng);
155 }

157 template <typename ChoiceBufferType>
158 double assess_momenta(const ChoiceBufferType& momenta) {
159  static double logSqrt2Pi = 0.5*std::log(2*M_PI);
160  double sum = 0.0;
161  for (const auto& momentum : momenta)
162  sum += -0.5 * momentum * momentum;
163  return sum - (momenta.cend() - momenta.cbegin()) * logSqrt2Pi;
164 }

166 template<typename Trace, typename Selection, typename RNG, typename ChoiceBuffer>
167 #ifdef __cpp_concepts
168 requires TopLevelGradientBasedChoiceUpdates<Trace,Selection,ChoiceBuffer> && Revertible<Trace>
169 #endif
170 bool hmc(Trace &trace, const Selection& selection,
171  size_t leapfrog_steps, double eps,
172  ChoiceBuffer& momenta_buffer,
173  ChoiceBuffer& values_buffer,
174  RNG &rng) {
175 
176  // NOTE: this read-only buffer is only valid up until the next call to update.
177  const ChoiceBuffer* gradient_buffer = &trace.choice_gradient(selection);
178 
179  // this overwrites the memory in the buffer
180  sample_momenta(momenta_buffer, rng);
181  double prev_momenta_score = assess_momenta(momenta_buffer);
182 
183  double log_weight = 0.0;
184  for (size_t step = 0; step < leapfrog_steps; step++) {
185 
186  // half step on momenta
187  momenta_buffer += (eps / 2.0) * (*gradient_buffer);
188 
189  // full step on positions
190  values_buffer += eps * momenta_buffer;
191 
192  // get incremental log weight and new gradient
193  double log_weight_increment = trace.update(
194  rng, gentl::change::no_change, values_buffer,
195  UpdateOptions().save(step == 0).precompute_gradient(true));
196  log_weight += log_weight_increment;
197  gradient_buffer = &trace.choice_gradient(selection);
198 
199  // half step on momenta
200  momenta_buffer += (eps / 2.0) * (*gradient_buffer);
201  }
202 
203  double new_momenta_score = assess_momenta(momenta_buffer);
204 
205  double prob_accept = std::min(1.0, std::exp(log_weight + new_momenta_score - prev_momenta_score));
206  bool accept = std::bernoulli_distribution{prob_accept}(rng);
207  if (!accept)
208  trace.revert();
209  return accept;
210 }
