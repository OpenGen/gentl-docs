28  double mh_accept_prob(double model_log_weight, double proposal_forward_score, double proposal_backward_score) {
29  return std::min(1.0, std::exp(model_log_weight + proposal_backward_score - proposal_forward_score));
30  }

32 // make_proposal must be a callable that takes a model trace as its only argument and returns a generative function
33 // with arguments instantiated
34  template<typename TraceType, typename ProposalTraceType,
35  typename MakeProposalType, typename ProposalParametersType,
36  typename RNGType>
37  bool mh(TraceType &model_trace, const MakeProposalType &make_proposal,
38  ProposalParametersType &proposal_params, RNGType &rng,
39  ProposalTraceType &proposal_trace, bool prepare_for_gradient=false) {
40  auto proposal = make_proposal(model_trace);
41  proposal.simulate(proposal_trace, rng, proposal_params, false);
42  double proposal_forward_score = proposal_trace.get_score();
43  const auto &change = proposal_trace.choices();
44  // note that log_weight and reverse_change and retdiff are only available in between
45  // a call to update and a call to revert (before calling update, they will error, and after calling revert
46  // they will error).
47  const auto& [model_log_weight, reverse_change, retdiff] = model_trace.update(
48  change, true, prepare_for_gradient);
49  auto backward_proposal = make_proposal(model_trace);
50  double proposal_backward_score = backward_proposal.assess(reverse_change, rng, proposal_params);
51  double prob_accept = mh_accept_prob(model_log_weight, proposal_forward_score, proposal_backward_score);
52  std::bernoulli_distribution dist{prob_accept};
53  bool accept = dist(rng);
54  if (!accept) {
55  model_trace.revert();
56  }
57  return accept;
58  }

60  template<typename TraceType, typename MakeProposalType, typename ProposalParametersType, typename RNGType>
61  size_t mh_chain(TraceType &model_trace, const MakeProposalType &make_proposal,
62  ProposalParametersType &proposal_params, size_t num_steps, RNGType &rng) {
63  // allocate initial proposal trace memory
64  auto proposal = make_proposal(model_trace);
65  auto proposal_trace = proposal.simulate(rng, proposal_params, false);
66  // do iterations, reusing proposal trace memory (as well as model trace memory)
67  size_t num_accepted = 0;
68  for (size_t step = 0; step < num_steps; step++) {
69  bool accepted = mh(model_trace, make_proposal, proposal_params, rng, proposal_trace);
70  if (accepted)
71  num_accepted++;
72  }
73  return num_accepted;
74  }

81  template <typename ChoiceBuffer, typename RNGType>
82  double mala_propose_values(ChoiceBuffer& proposal, const ChoiceBuffer& current,
83  const ChoiceBuffer& gradient, double tau, RNGType& rng) {
84  double stdev = std::sqrt(2 * tau);
85  static std::normal_distribution<double> standard_normal {0.0, 1.0};
86  static double logSqrt2Pi = 0.5*std::log(2*M_PI);
87  double log_density = (current.cend() - current.cbegin()) * (-std::log(stdev) - logSqrt2Pi);
88 
89  // first compute the mean in-place
90  proposal = current + (tau * gradient);
91 
92  // then sample new values (over-writing the mean), and finish computing the log density
93  for (auto proposal_it = proposal.begin(); proposal_it != proposal.end(); proposal_it++) {
94  double standard_normal_increment = standard_normal(rng);
95  *proposal_it += (standard_normal_increment * stdev);
96  log_density += -0.5 * (standard_normal_increment * standard_normal_increment);
97  }
98  return log_density;
99  }

101  template <typename ChoiceBuffer>
102  double mala_assess(const ChoiceBuffer& proposed, const ChoiceBuffer& current,
103  const ChoiceBuffer& gradient, double tau, ChoiceBuffer& storage) {
104  double stdev = std::sqrt(2 * tau);
105  static double logSqrt2Pi = 0.5*std::log(2*M_PI);
106  double log_density = (current.cend() - current.cbegin()) * (-std::log(stdev) - logSqrt2Pi);
107 
108  ChoiceBuffer& proposal_mean = storage; // rename it
109  proposal_mean = current + (tau * gradient);
110 
111  auto proposal_mean_it = proposal_mean.cbegin();
112  for (auto proposed_it = proposed.cbegin(); proposed_it != proposed.cend(); proposed_it++) {
113  double standard_normal_increment = (*proposed_it - *(proposal_mean_it++)) / stdev;
114  log_density += -0.5 * (standard_normal_increment * standard_normal_increment);
115  }
116  return log_density;
117  }

119  template<typename TraceType, typename SelectionType, typename RNGType, typename ChoiceBufferType>
120  bool mala(TraceType &trace, const SelectionType &selection, double tau,
121  ChoiceBufferType &storage1, ChoiceBufferType& storage2, RNGType &rng) {
122 
123  // NOTE: these buffers are only valid up until the next call to update.
124  ChoiceBufferType& proposed_values = storage1;
125  double forward_log_density = mala_propose_values(proposed_values, trace.choices(selection),
126  trace.choice_gradients(selection), tau, rng);
127  const auto& [log_weight, previous_values, retdiff] = trace.update(proposed_values, true, true);
128 
129  // compute backward log density
130  double backward_log_density = mala_assess(previous_values, proposed_values,
131  trace.choice_gradients(selection), tau,
132  storage2);
133 
134  double prob_accept = std::min(1.0, std::exp(log_weight + backward_log_density - forward_log_density));
135  std::bernoulli_distribution dist{prob_accept};
136  bool accept = dist(rng);
137  if (!accept) {
138  trace.revert();
139  }
140  return accept;
141  }

148  template <typename ChoiceBufferType, typename RNGType>
149  void sample_momenta(ChoiceBufferType& momenta, RNGType& rng) {
150  static std::normal_distribution<double> standard_normal{0.0, 1.0};
151  for (auto& momentum : momenta)
152  momentum = standard_normal(rng);
153  }

155  template <typename ChoiceBufferType>
156  double assess_momenta(const ChoiceBufferType& momenta) {
157  static double logSqrt2Pi = 0.5*std::log(2*M_PI);
158  double sum = 0.0;
159  for (const auto& momentum : momenta)
160  sum += -0.5 * momentum * momentum;
161  return sum - (momenta.cend() - momenta.cbegin()) * logSqrt2Pi;
162  }

164  template<typename TraceType, typename SelectionType, typename RNGType, typename ChoiceBufferType>
165  bool hmc(TraceType &trace, const SelectionType &selection,
166  size_t leapfrog_steps, double eps,
167  ChoiceBufferType &momenta_buffer,
168  ChoiceBufferType &values_buffer,
169  RNGType &rng) {
170 
171  // NOTE: this read-only buffer is only valid up until the next call to update.
172  const ChoiceBufferType* gradient_buffer = &trace.choice_gradients(selection);
173 
174  // this overwrites the memory in the buffer
175  sample_momenta(momenta_buffer, rng);
176  double prev_momenta_score = assess_momenta(momenta_buffer);
177 
178  double log_weight = 0.0;
179  for (size_t step = 0; step < leapfrog_steps; step++) {
180 
181  // half step on momenta
182  momenta_buffer += (eps / 2.0) * (*gradient_buffer);
183 
184  // full step on positions
185  values_buffer += eps * momenta_buffer;
186 
187  // get incremental log weight and new gradient
188  bool save_prev_state = (step == 0);
189  const auto&[log_weight_increment, backward_choices, retdiff] = trace.update(
190  values_buffer, save_prev_state, true);
191  log_weight += log_weight_increment;
192  gradient_buffer = &trace.choice_gradients(selection);
193 
194  // half step on momenta
195  momenta_buffer += (eps / 2.0) * (*gradient_buffer);
196  }
197 
198  double new_momenta_score = assess_momenta(momenta_buffer);
199 
200  double prob_accept = std::min(1.0, std::exp(log_weight + new_momenta_score - prev_momenta_score));
201  std::bernoulli_distribution dist{prob_accept};
202  bool accept = dist(rng);
203  if (!accept) {
204  trace.revert();
205  }
206  return accept;
207  }
