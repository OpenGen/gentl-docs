162 /*
163  * seed_seq_fe implements a fixed-entropy seed sequence; it conforms to all
164  * the requirements of a Seed Sequence concept.
165  *
166  * seed_seq_fe<N> implements a seed sequence which seeds based on a store of
167  * N * 32 bits of entropy. Typically, it would be initialized with N or more
168  * integers.
169  *
170  * seed_seq_fe128 and seed_seq_fe256 are provided as convenience typedefs for
171  * 128- and 256-bit entropy stores respectively. These variants outperform
172  * std::seed_seq, while being better mixing the bits it is provided as entropy.
173  * In almost all common use cases, they serve as better drop-in replacements
174  * for seed_seq.
175  *
176  * Technical details
177  *
178  * Assuming it constructed with M seed integers as input, it exhibits the
179  * following properties
180  *
181  * * Diffusion/Avalanche: A single-bit change in any of the M inputs has a
182  * 50% chance of flipping every bit in the bitstream produced by generate.
183  * Initializing the N-word entropy store with M words requires O(N * M)
184  * time precisely because of the avalanche requirements. Once constructed,
185  * calls to generate are linear in the number of words generated.
186  *
187  * * Bias freedom/Bijection: If M == N, the state of the entropy store is a
188  * bijection from the M inputs (i.e., no states occur twice, none are
189  * omitted). If M > N the number of times each state can occur is the same
190  * (each state occurs 2**(32*(M-N)) times, where ** is the power function).
191  * If M < N, some states cannot occur (bias) but no state occurs more
192  * than once (it's impossible to avoid bias if M < N; ideally N should not
193  * be chosen so that it is more than M).
194  *
195  * Likewise, the generate function has similar properties (with the entropy
196  * store as the input data). If more outputs are requested than there is
197  * entropy, some outputs cannot occur. For example, the Mersenne Twister
198  * will request 624 outputs, to initialize it's 19937-bit state, which is
199  * much larger than a 128-bit or 256-bit entropy pool. But in practice,
200  * limiting the Mersenne Twister to 2**128 possible initializations gives
201  * us enough initializations to give a unique initialization to trillions
202  * of computers for billions of years. If you really have 624 words of
203  * *real* high-quality entropy you want to use, you probably don't need
204  * an entropy mixer like this class at all. But if you *really* want to,
205  * nothing is stopping you from creating a randutils::seed_seq_fe<624>.
206  *
207  * * As a consequence of the above properties, if all parts of the provided
208  * seed data are kept constant except one, and the remaining part is varied
209  * through K different states, K different output sequences will be produced.
210  *
211  * * Also, because the amount of entropy stored is fixed, this class never
212  * performs dynamic allocation and is free of the possibility of generating
213  * an exception.
214  *
215  * Ideas used to implement this code include hashing, a simple PCG generator
216  * based on an MCG base with an XorShift output function and permutation
217  * functions on tuples.
218  *
219  * More detail at
220  * http://www.pcg-random.org/posts/developing-a-seed_seq-alternative.html
221  */

241  RANDUTILS_GENERALIZED_CONSTEXPR
242  static IntRep fast_exp(IntRep x, IntRep power)
243  {
244  IntRep result = IntRep(1);
245  IntRep multiplier = x;
246  while (power != IntRep(0)) {
247  IntRep thismult = power & IntRep(1) ? multiplier : IntRep(1);
248  result *= thismult;
249  power >>= 1;
250  multiplier *= multiplier;
251  }
252  return result;
253  }

317  template <typename InputIter>
318  void seed(InputIter begin, InputIter end, const std::vector<IntRep>& spawn_keys)
319  {
320  // concatenate the given entropy with the spawn keys
321  size_t entropy_len = std::distance(begin, end);
322  size_t spawn_keys_len = spawn_keys.size();
323  size_t entropy_and_spawn_keys_len = std::max(count, entropy_len) + spawn_keys_len;
324  std::vector<IntRep> entropy_and_spawn_keys(entropy_and_spawn_keys_len);
325  // begin with the provided entropy, padded with zeros up to length 'count'
326  auto dest = std::copy(begin, end, entropy_and_spawn_keys.begin());
327  for (size_t i = entropy_len; i < count; i++)
328  *dest++ = 0;
329  // fill in the rest with the spawn keys
330  dest = std::copy(spawn_keys.begin(), spawn_keys.end(), dest);
331  assert(dest == entropy_and_spawn_keys.end());
332 
333  mix_entropy(entropy_and_spawn_keys.begin(), entropy_and_spawn_keys.end());
334 
335  // For very small sizes, we do some additional mixing. For normal
336  // sizes, this loop never performs any iterations.
337  for (size_t i = 1; i < mix_rounds; ++i)
338  stir();
339  }
340 
341  seed_seq_fe& stir()
342  {
343  mix_entropy(mixer_.begin(), mixer_.end());
344  return *this;
345  }

349  template <size_t count, typename IntRep, size_t r>
350  template <typename InputIter>
351  void seed_seq_fe<count, IntRep, r>::mix_entropy(InputIter begin, InputIter end)
352  {
353  auto hash_const = INIT_A;
354  auto hash = [&](IntRep value) {
355  value ^= hash_const;
356  hash_const *= MULT_A;
357  value *= hash_const;
358  value ^= value >> XSHIFT;
359  return value;
360  };
361  auto mix = [](IntRep x, IntRep y) {
362  IntRep result = MIX_MULT_L*x - MIX_MULT_R*y;
363  result ^= result >> XSHIFT;
364  return result;
365  };
366 
367  InputIter current = begin;
368  for (auto& elem : mixer_) {
369  if (current != end)
370  elem = hash(*current++);
371  else
372  elem = hash(0U);
373  }
374  for (auto& src : mixer_)
375  for (auto& dest : mixer_)
376  if (&src != &dest)
377  dest = mix(dest,hash(src));
378  for (; current != end; ++current)
379  for (auto& dest : mixer_)
380  dest = mix(dest,hash(*current));
381  }

383  template <size_t count, typename IntRep, size_t mix_rounds>
384  template <typename OutputIterator>
385  void seed_seq_fe<count,IntRep,mix_rounds>::param(OutputIterator dest) const
386  {
387  const IntRep INV_A = fast_exp(MULT_A, IntRep(-1));
388  const IntRep MIX_INV_L = fast_exp(MIX_MULT_L, IntRep(-1));
389 
390  auto mixer_copy = mixer_;
391  for (size_t round = 0; round < mix_rounds; ++round) {
392  // Advance to the final value. We'll backtrack from that.
393  auto hash_const = INIT_A*fast_exp(MULT_A, IntRep(count * count));
394 
395  for (auto src = mixer_copy.rbegin(); src != mixer_copy.rend(); ++src)
396  for (auto dest = mixer_copy.rbegin(); dest != mixer_copy.rend();
397  ++dest)
398  if (src != dest) {
399  IntRep revhashed = *src;
400  auto mult_const = hash_const;
401  hash_const *= INV_A;
402  revhashed ^= hash_const;
403  revhashed *= mult_const;
404  revhashed ^= revhashed >> XSHIFT;
405  IntRep unmixed = *dest;
406  unmixed ^= unmixed >> XSHIFT;
407  unmixed += MIX_MULT_R*revhashed;
408  unmixed *= MIX_INV_L;
409  *dest = unmixed;
410  }
411  for (auto i = mixer_copy.rbegin(); i != mixer_copy.rend(); ++i) {
412  IntRep unhashed = *i;
413  unhashed ^= unhashed >> XSHIFT;
414  unhashed *= fast_exp(hash_const, IntRep(-1));
415  hash_const *= INV_A;
416  unhashed ^= hash_const;
417  *i = unhashed;
418  }
419  }
420  std::copy(mixer_copy.begin(), mixer_copy.end(), dest);
421  }

424  template <size_t count, typename IntRep, size_t mix_rounds>
425  template <typename RandomAccessIterator>
426  void seed_seq_fe<count,IntRep,mix_rounds>::generate(
427  RandomAccessIterator dest_begin,
428  RandomAccessIterator dest_end) const
429  {
430  auto src_begin = mixer_.begin();
431  auto src_end = mixer_.end();
432  auto src = src_begin;
433  auto hash_const = INIT_B;
434  for (auto dest = dest_begin; dest != dest_end; ++dest) {
435  auto dataval = *src;
436  if (++src == src_end)
437  src = src_begin;
438  dataval ^= hash_const;
439  hash_const *= MULT_B;
440  dataval *= hash_const;
441  dataval ^= dataval >> XSHIFT;
442  *dest = dataval;
443  }
444  }

482  if (sizeof(T) <= 4)

484  else {

504  default_seeds local_entropy()
505  {
506  // This is a constant that changes every time we compile the code
507  constexpr uint32_t compile_stamp =
508  fnv(2166136261U, __DATE__ __TIME__ __FILE__);
509 
510  // Some people think you shouldn't use the random device much because
511  // on some platforms it could be expensive to call or "use up" vital
512  // system-wide entropy, so we just call it once.
513  static uint32_t random_int = std::random_device{}();
514 
515  // The heap can vary from run to run as well.
516  void* malloc_addr = malloc(sizeof(int));
517  free(malloc_addr);
518  auto heap = hash(malloc_addr);
519  auto stack = hash(&malloc_addr);
520 
521  // Every call, we increment our random int. We don't care about race
522  // conditons. The more, the merrier.
523  random_int += 0xedf19156;
524 
525  // Classic seed, the time. It ought to change, especially since
526  // this is (hopefully) nanosecond resolution time.
527  auto hitime = std::chrono::high_resolution_clock::now()
528  .time_since_epoch().count();
529 
530  // Address of the thing being initialized. That can mean that
531  // different seed sequences in different places in memory will be
532  // different. Even for the same object, it may vary from run to
533  // run in systems with ASLR, such as OS X, but on Linux it might not
534  // unless we compile with -fPIC -pic.
535  auto self_data = hash(this);
536 
537  // The address of the time function. It should hopefully be in
538  // a system library that hopefully isn't always in the same place
539  // (might not change until system is rebooted though)
540  auto time_func = hash(&std::chrono::high_resolution_clock::now);
541 
542  // The address of the exit function. It should hopefully be in
543  // a system library that hopefully isn't always in the same place
544  // (might not change until system is rebooted though). Hopefully
545  // it's in a different library from time_func.
546  auto exit_func = hash(&_Exit);
547 
548  // The address of a local function. That may be in a totally
549  // different part of memory. On OS X it'll vary from run to run thanks
550  // to ASLR, on Linux it might not unless we compile with -fPIC -pic.
551  // Need the cast because it's an overloaded
552  // function and we need to pick the right one.
553  auto self_func = hash(
554  static_cast<uint32_t (*)(uint64_t)>(
555  &auto_seeded::crushto32));
556 
557  // Hash our thread id. It seems to vary from run to run on OS X, not
558  // so much on Linux.
559  auto thread_id = hash(std::this_thread::get_id());
560 
561  // Hash of the ID of a type. May or may not vary, depending on
562  // implementation.
563 #if __cpp_rtti || __GXX_RTTI
564  auto type_id = crushto32(typeid(*this).hash_code());
565 #else
566  uint32_t type_id = 0;
567 #endif
568 
569  // Platform-specific entropy
570  auto pid = crushto32(RANDUTILS_GETPID);
571  auto cpu = crushto32(RANDUTILS_CPU_ENTROPY);
572 
573  return {{random_int, crushto32(hitime), stack, heap, self_data,
574  self_func, exit_func, thread_id, type_id, pid, cpu}};
575  }
