54  template <typename T>
55  class General {
56  size_t min_t_;
57  std::unordered_map<size_t,T> constraints_;
58  public:
59  General() : min_t_(-1) {}
60  General& add(size_t t, T&& constraint) {
61  auto it = constraints_.find(t);
62  if (it == constraints_.end()) {
63  constraints_.insert({t, std::forward<T>(constraint)});
64  } else {
65  throw std::logic_error("cannot overwrite constraints");
66  }
67  min_t_ = std::min(min_t_, t);
68  }
69  size_t min_t() const {
70  return min_t_;
71  }
72  typename std::unordered_map<size_t,T>::const_iterator find(const size_t& key) const {
73  return constraints_.find();
74  }
75  typename std::unordered_map<size_t,T>::const_iterator cend() const {
76  return constraints_.cend();
77  }

143  // extend (does not mutate any state)
144  template<typename RNGType, typename ConstraintsType>
145  std::tuple<double, const constraints::Empty&, const valuechange::Unknown&> update(
146  RNGType& rng, const modelchange::Extend&, const constraints::NewStepOnly<ConstraintsType>& constraints,
147  bool save_previous, bool gradient) {
148  if (gradient)
149  throw std::logic_error("gradient not implemented");
150  if (save_previous)
151  throw std::logic_error("save_previous not yet implemented");
152  ValueType args = (t_ == 0) ? init_value_ : last_->subtrace().return_value();
153  auto [subtrace, log_weight] = submodel_(args).generate(constraints.constraints, rng, parameters_, false);
154  if (t_ == 0) {
155  last_ = std::make_shared<TimeSlice<SubtraceType>>(std::move(subtrace));
156  } else {
157  last_ = std::make_shared<TimeSlice<SubtraceType>>(std::move(subtrace), last_);
158  }
159  assert(last_->t() == t_);
160  t_++;
161  return {log_weight, constraints::empty, valuechange::unknown};
162  }

164  // mutates history (copy-on-write)
165  // TODO the trace owns the backwards object, but there are several types of backwards objects we could get
166  template<typename RNGType, typename ConstraintsType>
167  std::tuple<double, const constraints::General<ConstraintsType>&, const valuechange::Unknown&> update(
168  RNGType &rng, const modelchange::None&, const constraints::General<ConstraintsType>& constraints,
169  bool save_previous, bool gradient) {
170  if (gradient)
171  throw std::logic_error("gradient not implemented");
172  if (save_previous)
173  throw std::logic_error("save_previous not yet implemented");
174 
175  using slice_t = TimeSlice<SubtraceType>;
176 
177  // walk through the time slices backward to obtain a vector ('history') of pointers
178  // to all time slices between min_t (inclusive) and t_ (inclusive)
179  size_t min_t = constraints.min_t();
180  size_t history_length = t_ - min_t + 1;
181  std::vector<const slice_t*> history(history_length);
182  {
183  const slice_t *slice_ptr = last_.get();
184  for (auto it = history.rbegin(); it != history.rend(); it++) {
185  *it = slice_ptr;
186  slice_ptr = slice_ptr->prev().get();
187  }
188  }
189  assert(history.front()->t() == min_t);
190  assert(history.back()->t() == t_);
191 
192  // walk forward through the history, forking each trace and creating new time slices
193  std::shared_ptr<constraints::General<ConstraintsType>> backward;
194  std::shared_ptr<slice_t> prev {history.begin()->prev()};
195  std::variant<valuechange::Unknown,valuechange::None> input_change {valuechange::none};
196  double log_weight = 0.0;
197  for (const auto& slice_ptr : history) {
198  size_t t = slice_ptr->t();
199  auto forked_subtrace = slice_ptr->subtrace().fork();
200  auto constraints_it = constraints.find(t);
201  std::variant<constraints::Empty,ConstraintsType> sub_backward;
202  double sub_log_weight;
203  if (constraints_it == constraints.cend()) {
204  // did not find constraints, pass in empty constraints
205  std::tie(sub_log_weight, sub_backward, input_change) = forked_subtrace->update(
206  rng, input_change, constraints::empty, false, false);
207  if (!std::holds_alternative<constraints::empty>(sub_backward)) {
208  // this requirement could be relaxed
209  throw std::logic_error("backward constraint must have same type as forward constraints");
210  }
211  } else {
212  // found constraints
213  std::tie(sub_log_weight, sub_backward, input_change) = forked_subtrace->update(
214  rng, input_change, *constraints_it, false, false);
215  if (!std::holds_alternative<ConstraintsType>(sub_backward)) {
216  // this requirement could be relaxed
217  throw std::logic_error("backward constraint must have same type as forward constraints");
218  }
219  backward->add(t, sub_backward); // copy
220  }
221  log_weight += sub_log_weight;
222  prev = std::make_shared<TimeSlice<SubtraceType>>(std::move(forked_subtrace), std::move(prev));
223  }
224 
225  // store shared pointer to last step (increments reference count)
226  last_ = prev;
227 
228  backward_void_ptr_ = std::move(backward);
229  const auto& backward_ref = *static_cast<constraints::General<ConstraintsType>*>(backward_void_ptr_);
230  return {log_weight, backward_ref, valuechange::unknown};
231  }

234  return std::unique_ptr<Trace>(new Trace(*this));
235  }
