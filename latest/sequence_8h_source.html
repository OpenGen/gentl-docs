60  template <typename T>
61  class General {
62  size_t min_t_;
63  std::unordered_map<size_t,T> constraints_;
64  public:
65  General() : min_t_(-1) {}
66  General& add(size_t t, T&& constraint) {
67  auto it = constraints_.find(t);
68  if (it == constraints_.end()) {
69  constraints_.insert({t, std::forward<T>(constraint)});
70  } else {
71  throw std::logic_error("cannot overwrite constraints");
72  }
73  min_t_ = std::min(min_t_, t);
74  }
75  size_t min_t() const {
76  return min_t_;
77  }
78  typename std::unordered_map<size_t,T>::const_iterator find(const size_t& key) const {
79  return constraints_.find();
80  }
81  typename std::unordered_map<size_t,T>::const_iterator cend() const {
82  return constraints_.cend();
83  }
84  };

133 template <typename Submodel, typename Subtrace, typename Value, typename Parameters>
134 class Trace {
135 private:
136  size_t t_; // the number of time steps (0 means there are no subtraces)
137 // size_t last_private_t_; // the time step (starting from 0) of the last non-shared time slice
138  std::shared_ptr<TimeSlice<Subtrace>> last_; // initially a nullptr
139  const Submodel& submodel_;
140  Parameters& parameters_;
141  Value init_value_;
142  // void pointer because we don't know the type (see the ConstraintsType update implementation)
143  std::shared_ptr<void> backward_void_ptr_;
144  Trace(const Trace& other) = default;
145 public:
146  Trace(const Submodel& submodel, Value init_value, Parameters& parameters) :
147  t_(0),
148 // last_private_t_(0),
149  last_{nullptr},
150  submodel_(submodel), parameters_(parameters), init_value_(init_value),
151  backward_void_ptr_(nullptr) {}
152 
153  // extend (does not mutate any state)
154  template<typename RNGType, typename Constraint>
155  #ifdef __cpp_concepts
156  requires GeneratableGenerativeFunction<Submodel, Value, Parameters, Subtrace, Constraint>
157  #endif
158  double update(
159  RNGType& rng, const modelchange::Extend&, const constraints::NewStepOnly<Constraint>& constraints,
160  const UpdateOptions& options) {
161  if (options.precompute_gradient())
162  throw std::logic_error("gradient not implemented");
163  if (options.save())
164  throw std::logic_error("save_previous not yet implemented");
165  Value args = (t_ == 0) ? init_value_ : last_->subtrace().return_value();
166  auto [subtrace, log_weight] = submodel_(args).generate(rng, parameters_, constraints.constraints, GenerateOptions());
167  if (t_ == 0) {
168  last_ = std::make_shared<TimeSlice<Subtrace>>(std::move(subtrace));
169  } else {
170  last_ = std::make_shared<TimeSlice<Subtrace>>(std::move(subtrace), last_);
171  }
172  assert(last_->t() == t_);
173  t_++;
174  return log_weight;
175  }
176 
177  // mutates history (copy-on-write)
178  // TODO the trace owns the backwards object, but there are several types of backwards objects we could get
179  template<typename RNGType, typename Constraint>
180  #ifdef __cpp_concepts
181  requires Updatable<Subtrace, modelchange::None, Constraint>
182  #endif
183  std::tuple<double, const constraints::General<Constraint>&, const valuechange::Unknown&> update(
184  RNGType &rng, const modelchange::None&, const constraints::General<Constraint>& constraints,
185  const UpdateOptions& options) {
186  if (options.precompute_gradient())
187  throw std::logic_error("gradient not implemented");
188  if (options.save())
189  throw std::logic_error("save_previous not yet implemented");
190 
191  using slice_t = TimeSlice<Subtrace>;
192 
193  // walk through the time slices backward to obtain a vector ('history') of pointers
194  // to all time slices between min_t (inclusive) and t_ (inclusive)
195  size_t min_t = constraints.min_t();
196  size_t history_length = t_ - min_t + 1;
197  std::vector<const slice_t*> history(history_length);
198  {
199  const slice_t *slice_ptr = last_.get();
200  for (auto it = history.rbegin(); it != history.rend(); it++) {
201  *it = slice_ptr;
202  slice_ptr = slice_ptr->prev().get();
203  }
204  }
205  assert(history.front()->t() == min_t);
206  assert(history.back()->t() == t_);
207 
208  // walk forward through the history, forking each trace and creating new time slices
209  std::shared_ptr<constraints::General<Constraint>> backward;
210  std::shared_ptr<slice_t> prev {history.begin()->prev()};
211  std::variant<valuechange::Unknown,valuechange::None> input_change {valuechange::none};
212  double log_weight = 0.0;
213  for (const auto& slice_ptr : history) {
214  size_t t = slice_ptr->t();
215  auto forked_subtrace = slice_ptr->subtrace().fork();
216  auto constraints_it = constraints.find(t);
217  std::variant<constraints::Empty,Constraint> sub_backward;
218  double sub_log_weight;
219  if (constraints_it == constraints.cend()) {
220  // did not find constraints, pass in empty constraints
221  std::tie(sub_log_weight, sub_backward, input_change) = forked_subtrace->update(
222  rng, input_change, constraints::empty, false, false);
223  if (!std::holds_alternative<constraints::empty>(sub_backward)) {
224  // this requirement could be relaxed
225  throw std::logic_error("backward constraint must have same type as forward constraints");
226  }
227  } else {
228  // found constraints
229  std::tie(sub_log_weight, sub_backward, input_change) = forked_subtrace->update(
230  rng, input_change, *constraints_it, false, false);
231  if (!std::holds_alternative<Constraint>(sub_backward)) {
232  // this requirement could be relaxed
233  throw std::logic_error("backward constraint must have same type as forward constraints");
234  }
235  backward->add(t, sub_backward); // copy
236  }
237  log_weight += sub_log_weight;
238  prev = std::make_shared<TimeSlice<Subtrace>>(std::move(forked_subtrace), std::move(prev));
239  }
240 
241  // store shared pointer to last step (increments reference count)
242  last_ = prev;
243 
244  backward_void_ptr_ = std::move(backward);
245  const auto& backward_ref = *static_cast<constraints::General<Constraint>*>(backward_void_ptr_);
246  return {log_weight, backward_ref, valuechange::unknown};
247  }
248 
249  std::unique_ptr<Trace> fork() {
250  return std::unique_ptr<Trace>(new Trace(*this));
251  }
252 };
