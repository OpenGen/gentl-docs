32 template <typename Trace, typename RNGType>
33 #ifdef __cpp_concepts
34 requires Forkable<Trace>
35 #endif
36 class ParticleSystem {
37 private:
38  size_t num_particles_;
39  std::vector<std::unique_ptr<Trace>> traces_;
40  std::vector<std::unique_ptr<Trace>> traces_tmp_; // initially contains all nullptrs
41  std::vector<double> log_weights_;
42  std::vector<double> log_normalized_weights_;
43  std::vector<double> two_times_log_normalized_weights_;
44  std::vector<double> normalized_weights_;
45  std::vector<size_t> parents_;
46  std::vector<Trace*> trace_nonowning_ptrs_;
47  RNGType& rng_; // TODO this will need to be replaced with a seed_seq for the multi-threaded version.
48  double log_ml_estimate_;
49 
50  double normalize_weights() {
51  double log_total_weight = gentl::mathutils::logsumexp(log_weights_);
52  for (size_t i = 0; i < num_particles_; i++) {
53  log_normalized_weights_[i] = log_weights_[i] - log_total_weight;
54  two_times_log_normalized_weights_[i] = 2.0 * log_normalized_weights_[i];
55  normalized_weights_[i] = std::exp(log_normalized_weights_[i]);
56  }
57  return log_total_weight;
58  }
59 
60  void multinomial_resampling() {
61  std::discrete_distribution<size_t> dist(normalized_weights_.cbegin(), normalized_weights_.cend());
62  for (size_t i = 0; i < num_particles_; i++) {
63  parents_[i] = dist(rng_);
64  }
65  }
66 
67 public:
68  ParticleSystem(size_t num_particles, RNGType& rng) :
69  num_particles_(num_particles),
70  traces_(num_particles),
71  traces_tmp_(num_particles),
72  log_weights_(num_particles),
73  log_normalized_weights_(num_particles),
74  two_times_log_normalized_weights_(num_particles),
75  normalized_weights_(num_particles),
76  parents_(num_particles),
77  trace_nonowning_ptrs_(num_particles),
78  rng_(rng),
79  log_ml_estimate_(0.0) {
80  }

82  template <typename Model, typename Parameters, typename Constraints>
83  void init_step(const Model& model, Parameters& parameters, const Constraints& constraints) {
84  // TODO make prepare_for_gradient optional for each generate() and update() call
85  for (size_t i = 0; i < num_particles_; i++) {
86  auto [trace_ptr, log_weight] = model.generate(rng_, parameters, constraints, GenerateOptions());
87  traces_[i] = std::move(trace_ptr);
88  trace_nonowning_ptrs_[i] = traces_[i].get();
89  log_weights_[i] = log_weight;
90  }
91  }
92 
93  // TODO document requirements (and non-requirements) associated with normalization of the model
94  // (the model doesn't need to be normalized, and the normalizing constant can change)
95  // technically, this means you could just implement this via a ModelChange without constraints
96  // but the distinction is still useful for the common case when the models are normalized
97  template <typename InputChange, typename Constraints>
98  #ifdef __cpp_concepts
99  requires Updatable<Trace, InputChange, Constraints>
100  #endif
101  void step(const InputChange& input_change, const Constraints& constraints) {
102  for (size_t i = 0; i < traces_.size(); i++)
103  log_weights_[i] += traces_[i]->update(rng_, input_change, constraints, UpdateOptions());
104  }
105 
106  [[nodiscard]] double effective_sample_size() const {
107  return std::exp(-gentl::mathutils::logsumexp(two_times_log_normalized_weights_));
108  }
109 
110  double resample() {
111  // writes to normalized_weights_
112  double log_total_weight = normalize_weights();
113  log_ml_estimate_ += log_total_weight - std::log(num_particles_);
114 
115  // reads from normalized_weights_ and writes to parents_
116  multinomial_resampling();
117 
118  // TODO do this in a way that keeps traces in the same thread if you can to minimize data movement
119  // and avoids ruining the cache every time we resample
120  for (size_t i = 0; i < num_particles_; i++)
121  traces_tmp_[i] = traces_[parents_[i]]->fork();
122  for (size_t i = 0; i < num_particles_; i++) {
123  traces_[i] = std::move(traces_tmp_[i]); // move assignment
124  trace_nonowning_ptrs_[i] = traces_[i].get();
125  }
126  std::fill(log_weights_.begin(), log_weights_.end(), 0.0);
127  return log_total_weight;
128  }
129 
130  [[nodiscard]] double log_marginal_likelihood_estimate() const {
131  return log_ml_estimate_ + gentl::mathutils::logsumexp(log_weights_) - std::log(num_particles_);
132  }
133 
134  // so that user can run rejuvenation moves on them
135  const std::vector<Trace*>& traces() const {
136  return trace_nonowning_ptrs_;
137  }
138 
139 };
